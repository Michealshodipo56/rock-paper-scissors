JAVASCRIPT
let score = JSON.parse(localStorage.getItem('score'));
if (score === null){
  score = {
  win: 0,
  loose: 0,
  tie: 0,
}
}
show();

function option(hard){
  const move = picking();
  let result = '';
if (hard === 'rock'){
  if (move === 'rock'){
  result = 'tie';
} else if(move === 'paper'){
  result = 'you loose';
} else if (move === 'scissors'){
  result = 'you win';
}
} else if (hard === 'paper') {
  if (move === 'rock'){
  result = 'you win';
} else if(move === 'paper'){
  result = 'tie';
} else if (move === 'scissors'){
  result = 'you loose';
}
} else if(hard === 'scissors'){
  if (move === 'rock'){
  result = 'you loose';
} else if(move === 'paper'){
  result = 'you win';
} else if (move === 'scissors'){
  result = 'tie';
}
}

if (result === 'you win'){
  score.win += 1;
} else if (result === 'you loose'){
  score.loose += 1;
}else if (result === 'tie'){
  score.tie += 1;
}

localStorage.setItem('score', JSON.stringify(score));
show();

document.querySelector('.js-result')
.innerHTML = result;

document.querySelector('.js-move')
.innerHTML = `you picked <img class="move-icon" src="${hard}-emoji.png">
computer picked <img class="move-icon" src="${move}-emoji.png">`;
}

function show(){
document.querySelector('.js-score')
.innerHTML = `win: ${score.win}, loose:${score.loose}, tie: ${score.tie}.`;
}

function picking(){
  let random = Math.random();
  let move = '';
if (random >= 0 && random < 1/3){
  move = 'rock';
} else if (random >= 1/3 && random < 2/3){
  move = 'paper';
} else if (random >= 2/3 && random < 1){
  move = 'scissors';
}
return move; }
